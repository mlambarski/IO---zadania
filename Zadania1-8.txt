using System;
using System.Diagnostics;

namespace ConsoleApp5
{
    class Program
    {
        delegate int DelegateType(int n);
        static DelegateType fib_i;
        static DelegateType fib_r;
        static DelegateType sil_i;
        static DelegateType sil_r;

        public static int fib_iter(int n)
        {
            int a = 0;
            int b = 1;
            for (int i = 0; i < n; i++)
            {
                int temp = a;
                a = b;
                b = temp + b;
            }
            return a;
        }

        public static int fib_rek(int n)
        {
            if (n <= 1)
            {
                return 1;
            }
            else
            {
                return fib_rek(n - 2) + fib_rek(n - 1);
            }
        }


        public static int silnia_iter(int n)
        {
            int result = 1;
            for (int i = 1; i <= n; i++)
            {
                result *= i;
            }
            return result;
        }

        public static int silnia_rek(int n)
        {
            if (n < 1)
                return 1;
            else
                return n * silnia_rek(n - 1);
        }



        static void Main(string[] args)
        {
            long time1 = 0, time2 = 0, time3 = 0, time4 = 0;
            Stopwatch stopWatch1 = new Stopwatch();
            Stopwatch stopWatch2 = new Stopwatch();
            Stopwatch stopWatch3 = new Stopwatch();
            Stopwatch stopWatch4 = new Stopwatch();

            stopWatch1.Start();
            fib_i = new DelegateType(fib_iter);
            IAsyncResult fi = fib_i.BeginInvoke(10, null, null);
            stopWatch1.Stop();
            time1 = stopWatch1.ElapsedMilliseconds;
            Console.WriteLine("Fibonacci iteracyjnie: " + time1);

            stopWatch2.Start();
            fib_r = new DelegateType(fib_rek);
            IAsyncResult fr = fib_r.BeginInvoke(10, null, null);
            stopWatch2.Stop();
            time2 = stopWatch2.ElapsedMilliseconds;
            Console.WriteLine("Fibonacci rekurencyjnie: " + time2);

            stopWatch3.Start();
            sil_i = new DelegateType(silnia_iter);
            IAsyncResult si = sil_i.BeginInvoke(10, null, null);
            stopWatch3.Stop();
            time3 = stopWatch3.ElapsedMilliseconds;
            Console.WriteLine("Silnia iteracyjnie: " + time3);

            stopWatch4.Start();
            sil_r = new DelegateType(silnia_rek);
            IAsyncResult sr = sil_r.BeginInvoke(10, null, null);
            stopWatch4.Stop();
            time4 = stopWatch4.ElapsedMilliseconds;
            Console.WriteLine("Silnia rekurencyjnie: " + time4);

            Console.ReadKey();
        }
    }
}
